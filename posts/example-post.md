```rust
fn main() {
    println!("Hello from Rust!");
    let fibonacci = |n: u32| -> u32 {
        match n {
            0 => 0,
            1 => 1,
            _ => fibonacci(n - 1) + fibonacci(n - 2),
        }
    };
}
```
